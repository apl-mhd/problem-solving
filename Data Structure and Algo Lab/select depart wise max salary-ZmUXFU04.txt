FROM employees e1 WHERE NOT EXISTS( SELECT * FROM employees e2 WHERE e1.DEPARTMENT_ID = e2.DEPARTMENT_ID AND e1.SALARY<e2.SALARY )

SELECT e1.FIRST_NAME, d.DEPARTMENT_NAME
FROM employees e1 JOIN departments d on 
e1.DEPARTMENT_ID = d.DEPARTMENT_ID
 WHERE NOT EXISTS( SELECT * FROM employees e2 WHERE e1.DEPARTMENT_ID = e2.DEPARTMENT_ID AND 
              
                 e1.SALARY<e2.SALARY 
                );
